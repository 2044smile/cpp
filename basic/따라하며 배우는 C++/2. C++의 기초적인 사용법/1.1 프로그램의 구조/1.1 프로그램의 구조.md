# 1.1 프로그램의 구조

> https://www.youtube.com/watch?v=hzaeRckfL5w&list=PLNfg4W25Tapw5Yx4yuExHNybBIUk68aNz&index=10

```cpp
#include <iostream> // i(nput)o(utput) stream, #은 전처리기 (preprocessor directive)
#include <string>


int add(int a, int b)
{
	return a + b;
}


int main() // 프로그램 시작 함수
{
	(1 + 2) * (3 + 4); // 21; expression(값을 만들어내는 코드)
	int x = 2; // 리터럴: literal; 정보를 기록 memory 연산하는 cpu
	double b = 3.14;
	char c = 'A';
	bool d = true;
	int y = x + 3;

	int age = 30;
	double height = 181.7;

	// 출력
	std::cout << age << std::endl;
	std::cout << height << std::endl;

	// 입력
	std::cout << "Enter number: ";
	std::cin >> x;

	std::cout << x << std::endl;

	// 조건문
	if (x > 5) {
		std::cout << "x is big" << std::endl;
	}

	// 반복문
	for (int i = 0; i < 5; i++) {
		std::cout << i << std::endl;
	}

	// add 함수
	int result = add(3, 4);
	std::cout << result << std::endl << std::endl;

	// 문자열
	std::string name = "Tim";
	std::cout << name << std::endl;

	std::cout << y << std::endl; // namespace: 비슷한 기능들을 하는 함수들을 묶어 놓거나 이름이 겹칠 수 있을 때 네임스페이스를 사용한다
	std::cout << 1 + 1 << std::endl;
	// :: 네임스페이스 안에 들어있는
	// << 연산자, std: standard library

	return 0;  // return statement, ; 문장의 끝을 표현(반드시 있어야 한다)
}
// 함수는 내가 뭘 넣을테니까 함수에서 내부적으로 뭔가를 해서 결과를 달라
// 오퍼레이팅 시스템이 main() 이라는 함수(function)를 찾는다
// tip: ctrl + shift + b = build
// tip: ctrl + F5 = execute
```
